3. Suppose we were given the following assembly code. Let's say we know that before the execution of fun begins, %rsp stores 0x500.

	fun:
		pushq 0x5
		callq 0x1000	# Line 1
		callq foo	# Line 2
		ret

	0x1000:
		pushq 0x10
		cmpq %rip, 0x1000
		je .L1		# Line 3
		popq %rdx	# Line 4
		ret		# Line 5
	.L1:	
		pushq 0x15	
		ret		# Line 6		

	foo:
		popq %rax
		ret		# Line 7
